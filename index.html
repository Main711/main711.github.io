        async function callGeminiAPI(prompt, targetElement, maxRetries = 3) {

            targetElement.innerHTML = '<div class="flex justify-center items-center p-8"><div class="loader"></div></div>';

            openModal();

            const apiKey = ""; const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-05-20:generateContent?key=${apiKey}`;

            const payload = { contents: [{ parts: [{ text: prompt }] }] };

            for (let i = 0; i < maxRetries; i++) {

                try {

                    const response = await fetch(apiUrl, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(payload) });

                    if (!response.ok) throw new Error(`HTTP error! status: ${response.status}`);

                    const result = await response.json();

                    const candidate = result.candidates?.[0];

                    if (candidate && candidate.content?.parts?.[0]?.text) {

                        targetElement.innerHTML = candidate.content.parts[0].text.replace(/\n/g, '<br>');

                        return;

                    } else { throw new Error('Invalid response structure from API.'); }

                } catch (error) {

                    if (i === maxRetries - 1) {

                        targetElement.innerHTML = `<div class="p-4 bg-red-50 text-red-700 rounded-md"><p class="font-semibold">Ошибка</p><p>Не удалось получить ответ от API. Пожалуйста, попробуйте позже.</p></div>`;

                        return;

                    }

                    await new Promise(res => setTimeout(res, 1000 * 2 ** i));

                }

            }

        }

        function renderChart() {

            const { categories } = techDatabase[state.activeDomain];

            const labels = Object.values(categories).map(c => c.title);

            const data = Object.values(categories).map(c => c.technologies.length);

            if (categoryChart) categoryChart.destroy();

            categoryChart = new Chart(chartCanvasEl, {

                type: 'doughnut',

                data: {

                    labels,

                    datasets: [{ label: 'Количество технологий', data, backgroundColor: ['#38bdf8', '#0ea5e9', '#0284c7', '#0369a1'], borderColor: '#ffffff', borderWidth: 4, hoverOffset: 8 }]

                },

                options: {

                    responsive: true, maintainAspectRatio: false,

                    plugins: {

                        legend: { position: 'bottom', labels: { font: { family: 'Inter, sans-serif' }, padding: 15 } },

                        title: { display: true, text: 'Распределение технологий по категориям', font: { size: 16, weight: '600', family: 'Inter, sans-serif' }, padding: { bottom: 20 } }

                    },

                    cutout: '60%'

                }

            });

        }

 

        function updateView() {

        mainNavEl.addEventListener('click', e => {

            if (e.target.tagName === 'BUTTON') { state.activeDomain = e.target.dataset.domain; state.activeCategory = 'all'; updateView(); }

        });

 

        sidebarNavEl.addEventListener('click', e => {

            e.preventDefault(); if (e.target.tagName === 'A') { state.activeCategory = e.target.dataset.category; renderSidebar(); renderCards(); }

        });

 

        searchInputEl.addEventListener('input', e => { state.searchTerm = e.target.value; renderCards(); });

 

        cardsContainerEl.addEventListener('click', e => {

            const button = e.target.closest('.gemini-btn'); if (!button) return;

            const { action, title, description } = button.dataset;

            let prompt = '';

            if (action === 'explain') {

                modalTitleEl.textContent = `Объяснение: ${title}`;

                prompt = `Объясни технологию "${title}" простыми словами для новичка. Описание для контекста: ${description}. Расскажи о ключевых применениях. Ответ дай на русском языке в 1-2 абзацах.`;

            } else if (action === 'forecast') {

                modalTitleEl.textContent = `Прогноз для: ${title}`;

                prompt = `Сделай краткий прогноз будущих трендов для технологии "${title}" на следующие 5-10 лет. Описание для контекста: ${description}. Ответ дай на русском языке в виде короткого списка.`;

            }

            if(prompt) callGeminiAPI(prompt, modalBodyEl);

        });

 

        document.addEventListener('DOMContentLoaded', updateView);
